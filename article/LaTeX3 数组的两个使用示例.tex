	\section{数组的排序}
	
	\begin{lstlisting}
		\clist_set:Nn \l_foo_clist { 10, 7, 2, -2, 5, -6, 1 }
		\clist_sort:Nn \l_foo_clist
		{
			\int_compare:nNnTF { #1 } > { #2 }
			{ \sort_return_swapped: }
			{ \sort_return_same: }
		}
		
		\clist_map_inline:Nn \l_foo_clist {
			$#1$ \qquad 
		}
	\end{lstlisting}
	
	
	\ExplSyntaxOn
	\clist_set:Nn \l_foo_clist { 10, 7, 2, -2, 5, -6, 1 }

	}
	
	\clist_map_inline:Nn \l_foo_clist {
		$#1$ \qquad 
	}
